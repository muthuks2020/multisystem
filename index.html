Neo4j Knowledge Graph in Action

Sample Cypher Query

MATCH (p:Product)-[:BELONGS_TO]->(c:Category {name: 'running_shoes'})
WHERE p.price <= 100 AND p.size = 9

// Feature requirements
MATCH (p)-[f1:HAS_FEATURE]->(feat1:Feature {name: 'waterproof'})
WHERE f1.intensity >= 0.8

// Seasonal relevance
MATCH (p)-[:SUITABLE_FOR]->(s:Season {name: 'monsoon'})

// User preference matching
OPTIONAL MATCH (customer:Customer {id: 'sarah_j'})-[pref:PREFERS]->(feat:Feature)
WHERE (p)-[:HAS_FEATURE]->(feat)

RETURN p, match_score
ORDER BY match_score DESC
LIMIT 10
